void ChatWindow::onSendMessage() {
    QString text = m_inputEdit->text().trimmed();
    if (text.isEmpty()) {
        return;
    }

    // 创建用户消息
    Message userMsg;
    userMsg.content = text.toStdString();
    userMsg.type = MessageType::User;
    userMsg.timestamp = QDateTime::currentDateTime();

    // 添加到会话
    m_currentSession->addMessage(userMsg);
    appendMessage(userMsg);

    // 清空输入框
    m_inputEdit->clear();

    // 调用服务处理
    try {
        Message response = m_chatService->sendUserMessage(
            m_currentSession->getId(),
            userMsg.content
        );
        m_currentSession->addMessage(response);
        appendMessage(response);
    } catch (const std::exception& e) {
        qWarning() << "处理消息失败:" << e.what();
        statusBar()->showMessage(QString("处理失败: %1").arg(e.what()));
    }
}

// src/infrastructure/rendering/markdown_renderer.cpp
#include "markdown_renderer.h"
#include <cmark-gfm.h>
#include <QRegularExpression>
#include <QDebug>

MarkdownRenderer::MarkdownRenderer() {
    m_options = CMARK_OPT_DEFAULT | CMARK_OPT_GITHUB_PRE_LANG;
}

MarkdownRenderer::~MarkdownRenderer() {
}

RenderResult MarkdownRenderer::render(const std::string& markdown) {
    return doRender(markdown);
}

RenderResult MarkdownRenderer::doRender(const std::string& markdown) {
    RenderResult result;

    // 使用 cmark-gfm 解析
    char* html = cmark_markdown_to_html(
        markdown.c_str(),
        markdown.length(),
        m_options
    );

    if (!html) {
        throw std::runtime_error("Markdown 解析失败");
    }

    result.html = html;

    // 提取代码块
    QRegularExpression re("<pre><code.*?>(.*?)</code></pre>",
                        QRegularExpression::DotMatchesEverythingOption);
    QRegularExpressionMatchIterator i = re.globalMatch(QString::fromStdString(result.html));
    while (i.hasNext()) {
        QRegularExpressionMatch match = i.next();
        result.codeBlocks.push_back(match.captured(1).toStdString());
    }

    // 添加代码高亮
    result.html = addCodeHighlight(result.html);

    // 释放内存
    free(html);

    return result;
}

std::string MarkdownRenderer::addCodeHighlight(const std::string& html) {
    std::string result = html;

    // 简单的代码高亮（实际项目中可以使用 highlight.js）
    // 这里只做示例，替换常见的编程语言关键字
    std::vector<std::pair<std::string, std::string>> keywords = {
        {"class", "<span style='color: #E91E63'>class</span>"},
        {"public", "<span style='color: #E91E63'>public</span>"},
        {"private", "<span style='color: #E91E63'>private</span>"},
        {"int", "<span style='color: #2196F3'>int</span>"},
        {"std::string", "<span style='color: #4CAF50'>std::string</span>"},
        {"void", "<span style='color: #FF9800'>void</span>"}
    };

    for (const auto& pair : keywords) {
        size_t pos = 0;
        while ((pos = result.find(pair.first, pos)) != std::string::npos) {
            result.replace(pos, pair.first.length(), pair.second);
            pos += pair.second.length();
        }
    }

    return result;
}

std::shared_ptr<KnowledgeBase> KnowledgeLoader::loadKnowledgeBase(
    const std::string& name
) {
    auto kb = std::make_shared<KnowledgeBase>(name);

    QString dirPath = QString::fromStdString(m_knowledgeDir) + "/" + QString::fromStdString(name);
    QDir dir(dirPath);

    if (!dir.exists()) {
        throw std::runtime_error("知识库目录不存在: " + name);
    }

    // 加载所有 Markdown 文件
    auto documents = loadDocuments(dirPath.toStdString());
    for (const auto& doc : documents) {
        kb->addDocument(doc);
    }

    qDebug() << "加载知识库" << QString::fromStdString(name) << "完成，文档数:" << documents.size();

    return kb;
}

std::vector<Document> KnowledgeLoader::loadDocuments(const std::string& dirPath) {
    std::vector<Document> documents;

    QDir dir(QString::fromStdString(dirPath));
    QFileInfoList files = dir.entryInfoList(QStringList() << "*.md" << "*.markdown",
                                            QDir::Files | QDir::NoDotAndDotDot);

    for (const QFileInfo& fileInfo : files) {
        try {
            Document doc = loadDocument(fileInfo);
            documents.push_back(doc);
        } catch (const std::exception& e) {
            qWarning() << "加载文档失败:" << fileInfo.filePath() << e.what();
        }
    }

    return documents;
}

Document KnowledgeLoader::loadDocument(const QFileInfo& fileInfo) {
    Document doc;

    // 读取文件内容
    QFile file(fileInfo.filePath());
    if (!file.open(QIODevice::ReadOnly | QIODevice::Text)) {
        throw std::runtime_error("无法打开文件: " + fileInfo.filePath().toStdString());
    }

    QTextStream in(&file);
    QString content = in.readAll();
    file.close();

    // 设置文档信息
    doc.docId = DocumentId(fileInfo.absoluteFilePath().toStdString());
    doc.title = extractTitle(content.toStdString());
    doc.content = content.toStdString();
    doc.type = classifyDocument(fileInfo.absoluteFilePath().toStdString());

    return doc;
}

std::string KnowledgeLoader::extractTitle(const std::string& content) {
    // 尝试从 Markdown 中提取标题
    size_t pos = content.find("# ");
    if (pos != std::string::npos) {
        size_t end = content.find('\n', pos);
        if (end != std::string::npos) {
            return content.substr(pos + 2, end - pos - 2);
        }
    }

    // 如果没有找到标题，使用第一行
    pos = content.find('\n');
    if (pos != std::string::npos) {
        return content.substr(0, pos);
    }

    return "未命名文档";
}

KnowledgeType KnowledgeLoader::classifyDocument(const std::string& path) {
    std::string lowerPath = path;
    std::transform(lowerPath.begin(), lowerPath.end(), lowerPath.begin(), ::tolower);

    if (lowerPath.find("standard") != std::string::npos ||
        lowerPath.find("gb") != std::string::npos ||
        lowerPath.find("iso") != std::string::npos) {
        return KnowledgeType::DesignStandard;
    } else if (lowerPath.find("process") != std::string::npos ||
               lowerPath.find("flow") != std::string::npos) {
        return KnowledgeType::ProcessGuide;
    } else {
        return KnowledgeType::DrawingSpecification;
    }
}

std::vector<std::string> KnowledgeLoader::listAvailableKnowledgeBases() const {
    std::vector<std::string> bases;

    QDir dir(QString::fromStdString(m_knowledgeDir));
    QFileInfoList subdirs = dir.entryInfoList(QDir::Dirs | QDir::NoDotAndDotDot);

    for (const QFileInfo& subdir : subdirs) {
        bases.push_back(subdir.fileName().toStdString());
    }

    return bases;
}

```cpp
void GraphCanvas::paintEvent(QPaintEvent* event) {
    QPainter painter(this);
    painter.setRenderHint(QPainter::Antialiasing);

    // 应用缩放和平移
    painter.translate(m_offset);
    painter.scale(m_scale, m_scale);

    // 绘制边
    for (const auto& edge : m_edges) {
        // 查找源节点和目标节点
        auto fromNode = std::find_if(m_nodes.begin(), m_nodes.end(),
                                    [&](const GraphNode& n) { return n.id == edge.from; });
        auto toNode = std::find_if(m_nodes.begin(), m_nodes.end(),
                                  [&](const GraphNode& n) { return n.id == edge.to; });

        if (fromNode != m_nodes.end() && toNode != m_nodes.end()) {
            painter.setPen(QPen(QColor(200, 200, 200), edge.w));
            painter.drawLine(QPointF(fromNode->x, fromNode->y),
                          QPointF(toNode->x, toNode->y));
        }
    }

    // 绘制节点
    for (const auto& node : m_nodes) {
        QColor color(102, 126, 234);
        if (m_hoveredNode != -1 && m_nodes[m_hoveredNode].id == node.id) {
            color = color.lighter(120);  // 悬停高亮
        }

        painter.setBrush(QBrush(color));
        painter.setPen(Qt::NoPen);
        painter.drawEllipse(QPointF(node.x, node.y), node.r, node.r);

        // 绘制标签
        painter.setPen(QColor(50, 50, 50));
        painter.setFont(QFont("Arial", 9));
        painter.drawText(QPointF(node.x - 20, node.y + node.r + 15), node.label);
    }
}
```

void GraphCanvas::wheelEvent(QWheelEvent* event) {
    double delta = event->angleDelta().y() / 120.0;
    double newScale = m_scale * (1.0 + delta * 0.1);
    m_scale = qBound(0.3, newScale, 4.0);  // 限制缩放范围
    update();
}

void GraphCanvas::mousePressEvent(QMouseEvent* event) {
    if (event->button() == Qt::LeftButton) {
        m_dragging = true;
        m_dragStart = event->pos();
        m_dragOffset = m_offset;
    }
}

void GraphCanvas::mouseMoveEvent(QMouseEvent* event) {
    // 检测悬停节点
    QPointF pos = (event->pos() - m_offset) / m_scale;
    m_hoveredNode = -1;
    for (size_t i = 0; i < m_nodes.size(); ++i) {
        double dist = sqrt(pow(pos.x() - m_nodes[i].x, 2) +
                       pow(pos.y() - m_nodes[i].y, 2));
        if (dist < m_nodes[i].r) {
            m_hoveredNode = static_cast<int>(i);
            setToolTip(QString::fromStdString(m_nodes[i].label));
            break;
        }
    }

    // 拖拽平移
    if (m_dragging) {
        m_offset = m_dragOffset + (event->pos() - m_dragStart);
    }

    update();
}

void GraphCanvas::mouseReleaseEvent(QMouseEvent* event) {
    if (event->button() == Qt::LeftButton) {
        m_dragging = false;
    }
}

#include <QDebug>

SkillLoader::SkillLoader(const std::string& skillsDir)
    : m_skillsDir(skillsDir)
{
}

SkillLoader::SkillMetadata SkillLoader::loadMetadata(const std::string& skillPath) {
    QString metadataPath = QString::fromStdString(skillPath) + "/skill.json";
    QFile file(metadataPath);

    if (!file.open(QIODevice::ReadOnly)) {
        throw std::runtime_error("无法打开 skill.json: " + metadataPath.toStdString());
    }

    QJsonDocument doc = QJsonDocument::fromJson(file.readAll());
    file.close();

    if (!doc.isObject()) {
        throw std::runtime_error("skill.json 格式错误");
    }

    QJsonObject json = doc.object();

    SkillMetadata meta;
    meta.id = json["id"].toString().toStdString();
    meta.name = json["name"].toString().toStdString();
    meta.version = json["version"].toString().toStdString();
    meta.description = json["description"].toString().toStdString();
    meta.entry = json["entry"].toString().toStdString();
    meta.path = skillPath;

    return meta;
}

std::shared_ptr<ISkill> SkillLoader::loadSkillFromLibrary(const SkillMetadata& meta) {
    // 构建库文件路径
    QString libPath = QString::fromStdString(meta.path) + "/";

#ifdef Q_OS_WIN
    libPath += QString::fromStdString(meta.id) + ".dll";
#elif defined(Q_OS_LINUX)
    libPath += "lib" + QString::fromStdString(meta.id) + ".so";
#elif defined(Q_OS_MAC)
    libPath += "lib" + QString::fromStdString(meta.id) + ".dylib";
#else
    throw std::runtime_error("不支持的操作系统");
#endif

    QLibrary lib(libPath);
    if (!lib.load()) {
        throw std::runtime_error("无法加载 Skill 库: " + libPath.toStdString() +
                                ", 错误: " + lib.errorString().toStdString());
    }

    // 获取创建函数
    typedef ISkill* (*CreateFunc)();
    CreateFunc create = (CreateFunc)lib.resolve(meta.entry.c_str());
    if (!create) {
        lib.unload();
        throw std::runtime_error("无法找到 Skill 创建函数: " + meta.entry);
    }

    // 创建 Skill 实例
    ISkill* skill = create();
    if (!skill) {
        lib.unload();
        throw std::runtime_error("创建 Skill 实例失败");
    }

    // 返回智能指针，自定义删除器来卸载库
    return std::shared_ptr<ISkill>(skill, [lib](ISkill* s) mutable {
        delete s;
        lib.unload();
    });
}

std::shared_ptr<ISkill> SkillLoader::loadSkill(const std::string& skillId) {
    QString skillPath = QString::fromStdString(m_skillsDir) + "/" +
                       QString::fromStdString(skillId);

    QDir dir(skillPath);
    if (!dir.exists()) {
        throw std::runtime_error("Skill 目录不存在: " + skillId);
    }

    // 加载元数据
    SkillMetadata meta = loadMetadata(skillPath.toStdString());

    // 加载 Skill 实现
    return loadSkillFromLibrary(meta);
}

std::vector<std::string> SkillLoader::listAvailableSkills() const {
    std::vector<std::string> skills;

    QDir dir(QString::fromStdString(m_skillsDir));
    QFileInfoList subdirs = dir.entryInfoList(QDir::Dirs | QDir::NoDotAndDotDot);

    for (const QFileInfo& subdir : subdirs) {
        // 检查是否有 skill.json
        QString metadataPath = subdir.absoluteFilePath() + "/skill.json";
        if (QFile::exists(metadataPath)) {
            skills.push_back(subdir.fileName().toStdString());
        }
    }

    return skills;
}
