void ChatSession::addMessage(const Message& msg) {

    m_messages.push_back(msg);
}

std::vector<Message> ChatSession::getRecentMessages(int count) const {
    QMutexLocker locker(&m_mutex);

    std::vector<Message> result;
    int startIndex = std::max(0, (int)m_messages.size() - count);

    for (int i = startIndex; i < (int)m_messages.size(); ++i) {
        result.push_back(m_messages[i]);
    }

    return result;
}

void ChatSession::clearHistory() {
    QMutexLocker locker(&m_mutex);
    m_messages.clear();
}

    : m_id(KnowledgeId(name))
    , m_name(name)

void KnowledgeBase::addDocument(const Document& doc) {
    QMutexLocker locker(&m_mutex);
    m_documents.push_back(doc);
    buildSearchIndex();
}

void KnowledgeBase::removeDocument(const DocumentId& id) {
    QMutexLocker locker(&m_mutex);

    auto it = std::remove_if(m_documents.begin(), m_documents.end(),
        [&id](const Document& doc) {
            return doc.getId() == id;
        });
    m_documents.erase(it, m_documents.end());
    buildSearchIndex();
}

std::vector<Document> KnowledgeBase::searchDocuments(const std::string& keyword) const {
    QMutexLocker locker(&m_mutex);

    if (keyword.empty()) {
        return m_documents;
    }

    std::vector<Document> results;
    std::string lowerKeyword = keyword;
    std::transform(lowerKeyword.begin(), lowerKeyword.end(),
                  lowerKeyword.begin(), ::tolower);

    for (const auto& doc : m_documents) {
        std::string lowerTitle = doc.getTitle();
        std::string lowerContent = doc.getContent();

        std::transform(lowerTitle.begin(), lowerTitle.end(),
                      lowerTitle.begin(), ::tolower);
        std::transform(lowerContent.begin(), lowerContent.end(),
                      lowerContent.begin(), ::tolower);

        if (lowerTitle.find(lowerKeyword) != std::string::npos ||
            lowerContent.find(lowerKeyword) != std::string::npos) {
            results.push_back(doc);
        }
    }

    return results;
}

bool KnowledgeBase::isLoaded() const {
    QMutexLocker locker(&m_mutex);
    return !m_documents.empty();
}

std::vector<Document> KnowledgeBase::getAllDocuments() const {
    QMutexLocker locker(&m_mutex);
    return m_documents;
}

size_t KnowledgeBase::getDocumentCount() const {

#include "knowledge_loader.h"

#include <QFileInfo>
#include <QFile>
#include <QTextStream>
#include <QDebug>

KnowledgeLoader::KnowledgeLoader(const std::string& knowledgeDir)
    : m_knowledgeDir(knowledgeDir)
{
}

KnowledgeLoader::~KnowledgeLoader() {
}

std::shared_ptr<KnowledgeBase> KnowledgeLoader::loadKnowledgeBase(
    const std::string& name
) {
    auto kb = std::make_shared<KnowledgeBase>(name);

    QString dirPath = QString::fromStdString(m_knowledgeDir) + "/" +
                      QString::fromStdString(name);
    QDir dir(dirPath);

    if (!dir.exists()) {
        throw std::runtime_error("知识库目录不存在: " + name);
    }

    // 加载所有 Markdown 文件
    auto documents = loadDocuments(dirPath.toStdString());
    for (const auto& doc : documents) {
        kb->addDocument(doc);
    }

    qDebug() << "加载知识库" << QString::fromStdString(name)
             << "完成，文档数:" << documents.size();

    return kb;
}

std::vector<Document> KnowledgeLoader::loadDocuments(const std::string& dirPath) {
    std::vector<Document> documents;

    QDir dir(QString::fromStdString(dirPath));
    QFileInfoList files = dir.entryInfoList(
        QStringList() << "*.md" << "*.markdown",
        QDir::Files | QDir::NoDotAndDotDot
    );

    for (const QFileInfo& fileInfo : files) {
        try {
            Document doc = loadDocument(fileInfo.absoluteFilePath().toStdString());
            documents.push_back(doc);
        } catch (const std::exception& e) {
            qWarning() << "加载文档失败:" << fileInfo.filePath() << e.what();
        }
    }

    return documents;
}

Document KnowledgeLoader::loadDocument(const std::string& filePath) {
    Document doc;

    // 读取文件内容
    QFile file(QString::fromStdString(filePath));
    if (!file.open(QIODevice::ReadOnly | QIODevice::Text)) {
        throw std::runtime_error("无法打开文件: " + filePath);
    }

    QTextStream in(&file);
    QString content = in.readAll();
    file.close();

    // 设置文档信息
    doc.setId(DocumentId(filePath));
    doc.setTitle(extractTitle(content.toStdString()));
    doc.setContent(content.toStdString());
    doc.setType(classifyDocument(filePath));

    return doc;
}

std::string KnowledgeLoader::extractTitle(const std::string& content) {
    // 尝试从 Markdown 中提取标题
    size_t pos = content.find("# ");
    if (pos != std::string::npos) {
        size_t end = content.find('\n', pos);
        if (end != std::string::npos) {
            return content.substr(pos + 2, end - pos - 2);
        }
    }

    // 如果没有找到标题，使用第一行
    pos = content.find('\n');
    if (pos != std::string::npos) {
        return content.substr(0, pos);
    }

    return "未命名文档";
}

KnowledgeType KnowledgeLoader::classifyDocument(const std::string& path) {
    std::string lowerPath = path;
    std::transform(lowerPath.begin(), lowerPath.end(),
                  lowerPath.begin(), ::tolower);

    if (lowerPath.find("standard") != std::string::npos ||
        lowerPath.find("gb") != std::string::npos ||
        lowerPath.find("iso") != std::string::npos) {
        return KnowledgeType::DesignStandard;
    } else if (lowerPath.find("process") != std::string::npos ||
               lowerPath.find("flow") != std::string::npos) {
        return KnowledgeType::ProcessGuide;
    } else {
        return KnowledgeType::DrawingSpecification;
    }
}

std::vector<std::string> KnowledgeLoader::listAvailableKnowledgeBases() const {
    std::vector<std::string> bases;

    QDir dir(QString::fromStdString(m_knowledgeDir));
    QFileInfoList subdirs = dir.entryInfoList(QDir::Dirs | QDir::NoDotAndDotDot);

    for (const QFileInfo& subdir : subdirs) {
        bases.push_back(subdir.fileName().toStdString());
    }

    return bases;
}

Message ChatAppService::sendUserMessage(
    const SessionId& sessionId,
    const std::string& content
) {
    // 加载会话
    auto session = m_chatRepo->load(sessionId);
    if (!session) {
        throw std::runtime_error("会话不存在");
    }

    // 创建用户消息
    Message userMsg;
    userMsg.setContent(content);
    userMsg.setType(MessageType::User);
    userMsg.setTimestamp(QDateTime::currentDateTime());

    // 添加到会话
    session->addMessage(userMsg);

    // 处理消息
    ChatProcessor processor;
    Message assistantMsg = processor.processUserMessage(userMsg);

    // 添加回复到会话
    session->addMessage(assistantMsg);

    // 保存会话
    m_chatRepo->save(*session);

    return assistantMsg;
}

    auto session = m_chatRepo->load(sessionId);
    if (!session) {
        throw std::runtime_error("会话不存在");
    }

    return session->getRecentMessages(100);
}

    : m_dataDir(dataDir)
{
    // 创建必要的目录

    if (!dir.exists()) {

    }

    return getSessionsDir() + "/" + id.value + ".json";

    return getKnowledgeDir() + "/" + id.value + ".json";

void FileStorage::save(const ChatSession& session) {
    QJsonObject json;

    json["id"] = QString::fromStdString(session.getId().value);
    json["createdAt"] = session.getCreatedAt().toString(Qt::ISODate);

    QJsonArray messagesArray;
    for (const auto& msg : session.getAllDocuments()) {
        QJsonObject msgObj;
        msgObj["id"] = QString::fromStdString(msg.getId().value);
        msgObj["content"] = QString::fromStdString(msg.getContent());
        msgObj["type"] = static_cast<int>(msg.getType());
        msgObj["timestamp"] = msg.getTimestamp().toString(Qt::ISODate);
        messagesArray.append(msgObj);
    }
    json["messages"] = messagesArray;

    QJsonDocument doc(json);

    QString filePath = QString::fromStdString(getSessionFilePath(session.getId()));
    QFile file(filePath);
    if (!file.open(QIODevice::WriteOnly | QIODevice::Text)) {
        throw std::runtime_error("无法保存会话: " + filePath.toStdString());
    }

    file.write(doc.toJson());
    file.close();
}

std::shared_ptr<ChatSession> FileStorage::load(const SessionId& id) {
    QString filePath = QString::fromStdString(getSessionFilePath(id));
    QFile file(filePath);

    if (!file.exists()) {
        return nullptr;
    }

    if (!file.open(QIODevice::ReadOnly | QIODevice::Text)) {
        throw std::runtime_error("无法打开会话文件: " + filePath.toStdString());
    }

    QByteArray data = file.readAll();
    file.close();

    QJsonDocument doc = QJsonDocument::fromJson(data);
    QJsonObject json = doc.object();

    auto session = std::make_shared<ChatSession>(id);

    QJsonArray messagesArray = json["messages"].toArray();
    for (const auto& value : messagesArray) {
        QJsonObject msgObj = value.toObject();

        Message msg;
        msg.setContent(msgObj["content"].toString().toStdString());
        msg.setType(static_cast<MessageType>(msgObj["type"].toInt()));
        msg.setTimestamp(QDateTime::fromString(msgObj["timestamp"].toString(), Qt::ISODate));

        session->addMessage(msg);
    }

    return session;
}

std::vector<SessionId> FileStorage::listAllSessions() {
    std::vector<SessionId> ids;

    QDir dir(QString::fromStdString(getSessionsDir()));
    QStringList filters;
    filters << "*.json";
    QFileInfoList files = dir.entryInfoList(filters, QDir::Files);

    for (const QFileInfo& fileInfo : files) {
        std::string id = fileInfo.baseName().toStdString();
        ids.push_back(SessionId(id));
    }

    return ids;
}

void FileStorage::remove(const SessionId& id) {
    QString filePath = QString::fromStdString(getSessionFilePath(id));
    QFile::remove(filePath);
}

## 6. 算法设计

### 6.1 知识库搜索算法

**算法**: 关键词匹配搜索

```
输入: 关键词 keyword
输出: 匹配的文档列表

算法:
1. 将 keyword 转换为小写
2. 遍历所有文档
3. 对于每个文档:
   a. 将文档标题和内容转换为小写
   b. 检查标题或内容中是否包含 keyword
   c. 如果包含，则加入结果列表
4. 返回结果列表

时间复杂度: O(n * m)
其中 n 是文档数量，m 是平均文档长度
```

### 6.2 Markdown 渲染算法

**算法**: 使用 cmark-gfm 库

```
输入: Markdown 文本
输出: HTML 字符串

算法:
1. 调用 cmark_markdown_to_html() 解析 Markdown
2. 提取代码块
3. 对代码块添加语法高亮
4. 提取图片 URL
5. 返回 HTML

时间复杂度: O(n)
其中 n 是 Markdown 文本长度
```
